\begin{liturgicaltext}
    \priest Priest: For holy art Thou, O our God, and unto Thee do we send up glory: to the Father, and to the Son, and to the Holy Spirit, now and ever: \rubric{(and if a priest serve alone, he continues:)} and unto ages of ages.
\end{liturgicaltext}

\begin{rubricsection}

\begin{rubricmed}
    But if a deacon serve, the deacon exclaims:
\end{rubricmed}

\begin{liturgicaltext}
    \deacon O Lord, save the pious.
    \choir O Lord, save the pious.
    \deacon And hear us.
    \choir And hear us.
    \deacon And unto ages of ages.
\end{liturgicaltext}

\end{rubricsection}
